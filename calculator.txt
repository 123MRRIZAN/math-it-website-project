  <script>
    let display = document.getElementById("display");
    let steps = document.getElementById("steps");

    function appendVal(val) {
      display.value += val;
    }

    function clearDisplay() {
      display.value = "";
      steps.innerHTML = "";
    }

    function backspace() {
      display.value = display.value.slice(0, -1);
    }

    function calculate() {
      try {
        const expr = display.value;
        steps.innerHTML += "<div style='font-size:18px;'><b>Expression:</b> " + expr + "</div>";
        let simplified = math.simplify(expr);
        steps.innerHTML += "<div style='font-size:18px;'><b>Simplified:</b> " + simplified.toString() + "</div>";
        let result = math.evaluate(expr);
        steps.innerHTML += "<div style='font-size:18px;'><b>Result:</b> " + result + "</div><hr>";
        display.value = result;
      } catch {
        display.value = "Error";
      }
    }

    // Keyboard support
    document.addEventListener("keydown", (e) => {
      if (e.key === "Enter") calculate();
      else if (e.key === "Backspace") backspace();
    });

    // Tailwind button style
    document.querySelectorAll(".btn").forEach(btn => {
      btn.classList.add("bg-purple-600","text-white","px-10","py-3","rounded-lg","hover:bg-purple-800","transition");
    });
  </script>
